def clean_path(path)

  if(path.index(' '))
    return %{"#{path}"}
  end
  return path
end

def library
  # For some reason, my homepath returns inside 'My Documents'...
  application_data = File.join(home, LOCAL_SETTINGS, APPLICATION_DATA)
  if(File.exists?(application_data))
    return application_data
  else
    return super
  end
end
