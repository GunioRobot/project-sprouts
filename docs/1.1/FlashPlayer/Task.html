def execute *args
  super
  update_input_if_necessary
  execute_safely do
    update_mm_config
    update_trust_config_with input
  end

  if use_fdb?
    launch_fdb_and_player_with input
  else
    player_thread = launch_player_with input
    tail_flashlog player_thread
  end
end
